## Por que libass e não drawtext?

```
drawtext (NÃO usamos):               libass (USAMOS):
- Texto estático ou simples           - Formatação rica por palavra
- Sem highlight por palavra           - Efeito karaoke {\k} nativo
- Sem animações complexas             - Fontes embedded no container
- Posicionamento limitado             - Animações, bordas, sombras
- 1 cor por texto inteiro             - Multi-cor por caractere/palavra
```

**[CONFIRMADO]:** Análise de rede do Submagic não contém referências a Remotion, Three.js ou Canvas server-side. O método libass é o único viável para o estilo de captions deles.

## Formato ASS (Advanced SubStation Alpha)

### Estrutura do Arquivo

```ass
[Script Info]
Title: [NOME_DO_PRODUTO] Generated Captions
ScriptType: v4.00+
Collisions: Normal
PlayResX: 1080
PlayResY: 1920
Timer: 100.0000

[V4+ Styles]
Format: Name, Fontname, Fontsize, PrimaryColour, SecondaryColour, OutlineColour, BackColour, Bold, Italic, Underline, StrikeOut, ScaleX, ScaleY, Spacing, Angle, BorderStyle, Outline, Shadow, Alignment, MarginL, MarginR, MarginV, Encoding
Style: Default,Montserrat,36,&H00FFFFFF,&H000000FF,&H00000000,&H80000000,-1,0,0,0,100,100,0,0,1,2,1,2,10,10,60,1

[Events]
Format: Layer, Start, End, Style, Name, MarginL, MarginR, MarginV, Effect, Text
```

### Código de Cores ASS (BGR + Alpha)

```python
def hex_to_ass_color(hex_color: str, alpha: int = 0) -> str:
    """
    Converte cor hex para formato ASS: &HAABBGGRR
    alpha: 0=opaco, 255=transparente
    """
    hex_color = hex_color.lstrip("#")
    r = int(hex_color[0:2], 16)
    g = int(hex_color[2:4], 16)
    b = int(hex_color[4:6], 16)
    return f"&H{alpha:02X}{b:02X}{g:02X}{r:02X}"

# Exemplos:
# "#FFFFFF" (branco) → "&H00FFFFFF"
# "#FF6B00" (laranja Karl) → "&H00006BFF"
# transparente semi → "&H80000000"
```

## Gerador de Arquivo .ass

```python
# workers/ass_generator.py

from dataclasses import dataclass
from typing import List, Optional
import math

@dataclass
class CaptionTheme:
    font_family: str = "Montserrat"
    font_weight: int = 900
    font_size: int = 36
    text_color: str = "#FFFFFF"
    highlight_color: str = "#FF6B00"
    background_color: str = "transparent"
    shadow_enabled: bool = True
    shadow_color: str = "#000000"
    shadow_blur: int = 4
    outline_enabled: bool = True
    outline_color: str = "#000000"
    outline_width: int = 2
    position_x: int = 50   # % horizontal
    position_y: int = 55   # % vertical
    animation: Optional[str] = None
    emoji_enabled: bool = False

WORDS_PER_LINE = 3  # [INFERIDO: padrão viral de captions]

def generate_ass_file(
    words: List[dict],
    theme: CaptionTheme,
    video_width: int = 1080,
    video_height: int = 1920
) -> str:
    """
    Gera arquivo .ass com efeito karaoke word-by-word.
    [CONFIRMADO: método libass com {\\k} tags para highlight]
    """
    lines = group_words_into_caption_lines(words, WORDS_PER_LINE)

    margin_v = int(video_height * (1 - theme.position_y / 100))

    ass_text_color = hex_to_ass_color(theme.text_color)
    ass_highlight = hex_to_ass_color(theme.highlight_color)
    ass_outline = hex_to_ass_color(theme.outline_color if theme.outline_enabled else "#000000")
    ass_shadow = hex_to_ass_color(theme.shadow_color if theme.shadow_enabled else "#000000",
                                   alpha=0 if theme.shadow_enabled else 255)

    bold_flag = "-1" if theme.font_weight >= 700 else "0"
    outline_width = theme.outline_width if theme.outline_enabled else 0
    shadow_depth = 1 if theme.shadow_enabled else 0

    header = f"""[Script Info]
Title: Generated by [NOME_DO_PRODUTO]
ScriptType: v4.00+
Collisions: Normal
PlayResX: {video_width}
PlayResY: {video_height}
Timer: 100.0000

[V4+ Styles]
Format: Name, Fontname, Fontsize, PrimaryColour, SecondaryColour, OutlineColour, BackColour, Bold, Italic, Underline, StrikeOut, ScaleX, ScaleY, Spacing, Angle, BorderStyle, Outline, Shadow, Alignment, MarginL, MarginR, MarginV, Encoding
Style: Default,{theme.font_family},{theme.font_size},{ass_text_color},&H000000FF,{ass_outline},{ass_shadow},{bold_flag},0,0,0,100,100,0,0,1,{outline_width},{shadow_depth},2,10,10,{margin_v},1

[Events]
Format: Layer, Start, End, Style, Name, MarginL, MarginR, MarginV, Effect, Text
"""
    events = []
    for line in lines:
        active_words = [w for w in line if not w.get("isRemoved")]
        if not active_words:
            continue

        line_start = active_words[0]["startTime"]
        line_end = active_words[-1]["endTime"]

        # Construir texto com tags karaoke {\\k} por palavra
        # {\\k N} = duração em centisegundos até a próxima palavra começar
        # Enquanto a palavra está "ativa", fica com highlight_color

        text_parts = []
        for i, word in enumerate(active_words):
            if word["type"] != "word":
                continue

            word_duration_cs = int((word["endTime"] - word["startTime"]) * 100)

            # Tag de karaoke: enquanto ativo = highlight_color
            # {\\kf N} = karaoke fill (preenchimento progressivo da esquerda)
            # {\\k N} = karaoke simples (troca instantânea de cor)
            karaoke_tag = f"{{\\kf{word_duration_cs}}}"

            # Cor de destaque para palavra ativa = highlight_color
            # Palavras futuras ficam em text_color (cor padrão do style)
            text_parts.append(f"{karaoke_tag}{word['text']}")

            # Espaço entre palavras (exceto última)
            if i < len(active_words) - 1:
                text_parts.append(" ")

        line_text = "".join(text_parts)

        # Timing no formato ASS: H:MM:SS.cs
        start_ts = format_ass_timestamp(line_start)
        end_ts = format_ass_timestamp(line_end)

        events.append(
            f"Dialogue: 0,{start_ts},{end_ts},Default,,0,0,0,,{line_text}"
        )

    return header + "\n".join(events) + "\n"


def format_ass_timestamp(seconds: float) -> str:
    """Converte segundos para formato H:MM:SS.cs"""
    h = int(seconds // 3600)
    m = int((seconds % 3600) // 60)
    s = int(seconds % 60)
    cs = int((seconds % 1) * 100)
    return f"{h}:{m:02d}:{s:02d}.{cs:02d}"


def group_words_into_caption_lines(words: list, words_per_line: int) -> list:
    """Agrupa palavras em linhas de caption"""
    lines = []
    current_line = []
    word_count = 0

    for word in words:
        if word.get("isRemoved"):
            continue

        current_line.append(word)

        if word["type"] == "word":
            word_count += 1
            if word_count >= words_per_line:
                lines.append(current_line)
                current_line = []
                word_count = 0

    if current_line:
        lines.append(current_line)

    return lines
```

## FFmpeg filter_complex Completo

```python
# workers/ffmpeg_render.py

import subprocess
from typing import List, Optional

def run_ffmpeg_render(
    input_path: str,
    ass_path: str,
    output_path: str,
    cuts: List[dict],           # [{start, end}] silences/bad takes removidos
    broll_items: List[dict],    # [{assetUrl, startTime, endTime, layout}]
    music_config: Optional[dict],# {localPath, volume, startFromTime, fade}
    hook_title: Optional[dict], # {text, template, top, size}
    magic_zooms: List[dict],    # [{startTime, endTime, zoomFactor}]
    resolution: str = "1080p",
    video_width: int = 1080,
    video_height: int = 1920,
    fonts_dir: str = "/fonts"
):
    """
    Monta e executa o FFmpeg filter_complex completo.
    [CONFIRMADO: FFmpeg + libass — método confirmado da indústria]
    """
    inputs = ["-i", input_path]
    filter_parts = []
    current_stream = "[0:v]"
    audio_stream = "[0:a]"
    input_index = 1

    # ── 1. Remoção de silêncios/bad takes ──────────────────────────
    # Usa concat filter com timestamps de keeps (inverso dos cuts)
    if cuts:
        keeps = compute_keeps(cuts, get_video_duration(input_path))
        concat_v_parts = []
        concat_a_parts = []
        for i, keep in enumerate(keeps):
            label_v = f"[trim{i}v]"
            label_a = f"[trim{i}a]"
            filter_parts.append(
                f"[0:v]trim={keep['start']:.3f}:{keep['end']:.3f},"
                f"setpts=PTS-STARTPTS{label_v}"
            )
            filter_parts.append(
                f"[0:a]atrim={keep['start']:.3f}:{keep['end']:.3f},"
                f"asetpts=PTS-STARTPTS{label_a}"
            )
            concat_v_parts.append(label_v)
            concat_a_parts.append(label_a)

        n = len(keeps)
        concat_input = "".join(concat_v_parts + concat_a_parts)
        filter_parts.append(f"{concat_input}concat=n={n}:v=1:a=1[trimmed_v][trimmed_a]")
        current_stream = "[trimmed_v]"
        audio_stream = "[trimmed_a]"

    # ── 2. Burn-in de captions via libass ──────────────────────────
    # [CONFIRMADO: ffmpeg -vf "ass=captions.ass:fontsdir=/fonts/"]
    filter_parts.append(
        f"{current_stream}ass='{ass_path}':fontsdir='{fonts_dir}'[captioned]"
    )
    current_stream = "[captioned]"

    # ── 3. Magic Zooms (zoompan filter) ────────────────────────────
    if magic_zooms:
        for i, zoom in enumerate(magic_zooms):
            label = f"[zoom{i}]"
            duration = zoom["endTime"] - zoom["startTime"]
            fps = 30
            n_frames = int(duration * fps)
            factor = zoom.get("zoomFactor", 1.3)
            filter_parts.append(
                f"{current_stream}zoompan="
                f"z='if(between(t,{zoom['startTime']},{zoom['endTime']}),{factor},1)':"
                f"d={n_frames}:s={video_width}x{video_height}"
                f"{label}"
            )
            current_stream = label

    # ── 4. B-roll overlay ──────────────────────────────────────────
    # [CONFIRMADO: layouts: cover | contain | pip-top-right | split-50-50]
    for i, item in enumerate(broll_items):
        inputs.extend(["-i", item["localPath"]])
        broll_stream = f"[{input_index}:v]"
        input_index += 1

        overlay_expr = compute_broll_overlay(
            item["layout"], video_width, video_height,
            item["startTime"], item["endTime"]
        )

        label = f"[broll{i}]"
        filter_parts.append(
            f"{broll_stream}scale={overlay_expr['scale']}[bscaled{i}];"
            f"{current_stream}[bscaled{i}]overlay="
            f"x={overlay_expr['x']}:y={overlay_expr['y']}:"
            f"enable='between(t,{item['startTime']},{item['endTime']})'"
            f"{label}"
        )
        current_stream = label

    # ── 5. Hook Title (drawtext sobre o .ass) ──────────────────────
    if hook_title:
        top_pct = hook_title.get("top", 45) / 100
        y_pos = int(video_height * top_pct)
        font_size = hook_title.get("size", 32)
        text = hook_title["text"].replace("'", "\\'")

        label = "[withhook]"
        filter_parts.append(
            f"{current_stream}drawtext="
            f"fontfile='{fonts_dir}/Montserrat-Black.ttf':"
            f"text='{text}':"
            f"fontsize={font_size}:"
            f"fontcolor=white:"
            f"x=(w-text_w)/2:y={y_pos}:"
            f"shadowcolor=black:shadowx=2:shadowy=2"
            f"{label}"
        )
        current_stream = label

    filter_parts.append(f"{current_stream}[vout]")

    # ── 6. Mix de áudio (música de fundo) ──────────────────────────
    if music_config and music_config.get("localPath"):
        inputs.extend(["-i", music_config["localPath"]])
        volume = music_config.get("volume", 30) / 100
        fade = music_config.get("fade", True)
        start_time = music_config.get("startFromTime", 0)

        fade_filter = f",afade=t=out:st=__DURATION__:d=2" if fade else ""
        filter_parts.append(
            f"[{input_index}:a]"
            f"atrim=start={start_time},"
            f"asetpts=PTS-STARTPTS,"
            f"volume={volume}"
            f"{fade_filter}"
            f"[music];"
            f"{audio_stream}[music]amix=inputs=2:duration=first:dropout_transition=2[aout]"
        )
        audio_out = "[aout]"
        input_index += 1
    else:
        audio_out = audio_stream

    # ── Montar comando final ────────────────────────────────────────
    filter_complex = ";".join(filter_parts)

    cmd = [
        "ffmpeg", "-y",
        *inputs,
        "-filter_complex", filter_complex,
        "-map", "[vout]",
        "-map", audio_out,
        "-c:v", "libx264",
        "-crf", "18",
        "-preset", "fast",
        "-profile:v", "high",
        "-level", "4.0",
        "-c:a", "aac",
        "-b:a", "128k",
        "-movflags", "+faststart",  # streaming-ready
        "-pix_fmt", "yuv420p",
        output_path
    ]

    result = subprocess.run(cmd, capture_output=True, text=True, timeout=600)

    if result.returncode != 0:
        raise RuntimeError(f"FFmpeg error: {result.stderr}")

    return output_path
```

## Compute B-Roll Overlay por Layout

```python
def compute_broll_overlay(
    layout: str,
    vw: int, vh: int,
    start_time: float, end_time: float
) -> dict:
    """
    Calcula parâmetros de overlay para cada layout.
    [CONFIRMADO: layouts cover|contain|pip-top-right|split-50-50 etc.]
    """
    layouts = {
        "cover": {
            "scale": f"{vw}:{vh}",
            "x": "0", "y": "0"
        },
        "contain": {
            "scale": f"{vw}:{vh}:force_original_aspect_ratio=decrease",
            "x": "(W-w)/2", "y": "(H-h)/2"
        },
        "pip_top_right": {
            "scale": f"{int(vw * 0.35)}:{int(vh * 0.25)}",
            "x": f"{int(vw * 0.62)}", "y": f"{int(vh * 0.03)}"
        },
        "pip_bottom_right": {
            "scale": f"{int(vw * 0.35)}:{int(vh * 0.25)}",
            "x": f"{int(vw * 0.62)}", "y": f"{int(vh * 0.72)}"
        },
        "split_50_50": {
            "scale": f"{vw}:{int(vh * 0.5)}",
            "x": "0", "y": "0"
        },
        "split_35_65": {
            "scale": f"{vw}:{int(vh * 0.35)}",
            "x": "0", "y": "0"
        },
        "rounded": {
            "scale": f"{int(vw * 0.9)}:{int(vw * 0.9)}",
            "x": f"{int(vw * 0.05)}", "y": f"{int(vh * 0.15)}"
        },
        "square": {
            "scale": f"{vw}:{vw}",
            "x": "0", "y": f"{int((vh - vw) / 2)}"
        },
    }
    return layouts.get(layout, layouts["cover"])
```
